177 void RDLCDWrite(uint8_t byte, uint8_t dc) {
178  // Set data/command mode
179  RDLCD_PORT = (RDLCD_PORT & ~(1 << RDLCD_DC)) | (dc << RDLCD_DC);
180  // Write byte to SPI
181  RDSPIRWByte(byte, 5, &RDLCD_PORT, RDLCD_CS);
182 }

187 void RDLCDInit(void) {
188  // Init LCD control pins, pull CS and RST high
189  DDRC |= (1 << RDLCD_CS) | (1 << RDLCD_RST) | (1 << RDLCD_DC);
190  RDLCD_PORT |= (1 << RDLCD_CS) | (1 << RDLCD_RST);
191 
192  // Reset LCD
193  RDLCD_PORT &= ~(1 << RDLCD_RST);
194  RDLCD_PORT |= (1 << RDLCD_RST);
195 
196  // Init SPI - little endian, CPOL = 0, CPHA = 0
197  RDSPIInit(0,0);
198  // Use extended instruction set
199  RDLCDWrite(LCD_EXT_FN, RDLCD_C);
200  // Set default contrast
201  RDLCDWrite(LCD_SET_CONTRAST | RDLCD_HIGH_CONTRAST, RDLCD_C);
202  // Use basic instruction set
203  RDLCDWrite(LCD_BSC_FN, RDLCD_C);
204  //Set LCD to normal mode
205  RDLCDWrite(LCD_NORM, RDLCD_C);
206 
207  //Reset the LCD position
208  RDLCDWrite(LCD_SET_X | 0x00, RDLCD_C);
209  RDLCDWrite(LCD_SET_Y | 0x00, RDLCD_C);
210 
211 }

228 void RDLCDCharacter(unsigned char character) {
229  // Clear padding between each character
230  RDLCDWrite(0x00, RDLCD_D);
231  for (int i = 0; i < RDLCD_FONT_W; i++)
232  {
233  RDLCDWrite(ASCII[character - 0x20][i], RDLCD_D);
234  }
235  RDLCDWrite(0x00, RDLCD_D);
236 }
