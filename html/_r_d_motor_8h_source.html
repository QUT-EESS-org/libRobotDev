45 void RDTimerInit(void) {
46  set_bit(TCCR1A, COM1A1); // Compare Match A: Clear on match, Set at TOP.
47  set_bit(TCCR1A, COM1B1); // Compare Match B: Clear on match, Set at TOP.
48  set_bit(TCCR1B, WGM13); set_bit(TCCR1B, WGM12); set_bit(TCCR1A, WGM11); // Set 16bit fast PWM. TOP at ICR.
49  set_bit(TCCR1B, CS11); set_bit(TCCR1B, CS10); // /64 prescaler.
50  ICR1 = 0x0FFF; // Set TOP value for 12bit resolution.
51  TCNT1 = 0x00; // Set counter to 0.
52 
53  set_bit(TCCR3A, COM3A1); // Compare Match A: Clear on match, Set at TOP.
