192 static unsigned long RDBluetoothReturnBaudUL(char baud) {
193  // Convert character to index value (0 : 12)
194  uint8_t baudVal = (baud < 'A') ? baud - '0' : (baud - 'A') + 10;
195  /*
196  * 1 - 6 : 1200 * 2^(baudVal - 1)
197  * 7 - 11 : 1200 * 2^(baudVal - 1) - 19200 * 2^(baudVal - 7)
198  * 12 : 1382400
199  */
200  return (baudVal >= 12) ? 1382400 : 1200 * (1 << (baudVal - 1)) - (baudVal >= 7) * ( 19200 * (1 << (baudVal - 7)) );
201 }
