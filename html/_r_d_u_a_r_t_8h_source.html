77 void RDUARTInit(unsigned long baud)
78 {
79  // Turn off interrupts (necessary when setting UART).
80  cli();
81  // Resets the tail and headers for both input and output buffers
82  outputBuffer.bufferTail = outputBuffer.bufferHead = 0;
83  inputBuffer.bufferTail = inputBuffer.bufferHead = 0;
84  // Dynamically allocates space for both input and output data buffers
85  outputBuffer.dataBuffer = (uint8_t*) malloc(OUTPUT_BUFFER_SIZE * sizeof(uint8_t));
86  inputBuffer.dataBuffer = (uint8_t*) malloc(INPUT_BUFFER_SIZE * sizeof(uint8_t));
87  // Set the baud rate for the device.
88  //UBRR1 = (F_CPU / (8 * baud)) - 1; Apparently this doesn't work - Tested with arduino
89  UBRR1 = ((F_CPU / (4 * baud)) - 1)/2; // Apparently this does work :(
90  // Sets the control & status register A to use double transmission speeds.
91  UCSR1A = (1 << U2X1);
92  // Sets the control & status register B to enable transmitting and
93  // receiving. Transmitting and receiving interrupts is also enabled.
94  UCSR1B = (1 << RXCIE1);
95  UCSR1B |= (1 << RXEN1) | (1 << TXEN1);
96  // Sets the control & status register C. Parity bits are disabled and only 1 stop bit used.
97  // 8 bits will be used for transferral (UCSZn1 and UCSZn0 are both set to 1)
98  UCSR1C = (3 << UCSZ10);
99  // Turn on interrupts
100  sei();
101 }
