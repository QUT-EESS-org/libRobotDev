100 uint16_t RDAnalogRead(unsigned char channel, unsigned char mode) {
101  ADMUX |= channel; // Channel to be read
102  if (mode == MODE_8_BIT) {
103  set_bit(ADMUX, ADLAR); // Left align if 8-bit mode
104  } else {
105  clr_bit(ADMUX, ADLAR); // Right align if 10-bit mode
106  }
107 
108  set_bit(ADCSRA, ADSC); // Start conversion
109  while (get_bit(ADCSRA, ADSC)) { ; } // Wait for conversion to finish
110  // Return value mode-specific
111  if (mode == MODE_8_BIT) {
112  return ADCH;
113  } else {
114  return ADC;
115  }
116 }

173 void RDAnalogReadCont(unsigned char channel, unsigned char enable) {
174  ADMUX |= channel;
175  set_bit(ADCSRA, ADATE); // Auto trigger enable, ADTS bits 0 for free running mode
